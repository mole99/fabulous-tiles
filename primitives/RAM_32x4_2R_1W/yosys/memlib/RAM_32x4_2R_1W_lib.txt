# Yosys doesn't support configurable sync/async ports.
# So we define three RAMs for 2xasync, 1xsync 1xasync and 2xsync

ram distributed $__RAM_32x4_2R_1W_AA_ {
    abits 5;
    width 4;
    cost 6;
    port sw "A" {
        clock anyedge; # PORT_<name>_CLK_POL
    }
    port ar "B" {
    }
    port ar "C" {
    }
}

ram distributed $__RAM_32x4_2R_1W_SA_ {
    abits 5;
    width 4;
    cost 5;
    port sw "A" {
        clock anyedge; # PORT_<name>_CLK_POL
        wrtrans all old;
    }
    port sr "B" {
        clock anyedge; # PORT_<name>_CLK_POL
        
        # Has a read clock enable signal
        rden; # PORT_<name>_RD_EN
    }
    port ar "C" {
    }
}

ram distributed $__RAM_32x4_2R_1W_SS_ {
    abits 5;
    width 4;
    cost 4;
    port sw "A" {
        clock anyedge; # PORT_<name>_CLK_POL
        wrtrans all old;
    }
    port sr "B" {
        clock anyedge; # PORT_<name>_CLK_POL
        
        # Has a read clock enable signal
        rden; # PORT_<name>_RD_EN
    }
    port sr "C" {
        clock anyedge; # PORT_<name>_CLK_POL
        
        # Has a read clock enable signal
        rden; # PORT_<name>_RD_EN
    }
}
